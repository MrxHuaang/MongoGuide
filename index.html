01 — Operaciones Básicas

CRUD Completo

Create, Read, Update, Delete — las cuatro operaciones fundamentales.

➕ INSERT — Crear documentos
mongo shell
// ── insertOne ─────────────────────────────────
// Inserta exactamente UN documento
db.productos.insertOne({
  nombre: "Smartphone X",
  precio: 799,
  stock: 100,
  categoria: "tecnologia"
})
// Retorna: { acknowledged: true, insertedId: ObjectId(...) }

// ── insertMany ────────────────────────────────
// Inserta un array de documentos
db.productos.insertMany([
  { nombre: "Cable HDMI",   precio: 12, stock: 500 },
  { nombre: "Cargador USB", precio: 20, stock: 300 }
])
// Retorna: { acknowledged: true, insertedIds: { 0: ..., 1: ... } }

// ── insertMany con ordered: false ─────────────
// Si uno falla, continúa con los demás (no detiene todo)
db.productos.insertMany(
  [
    { _id: 1, nombre: "A" },
    { _id: 1, nombre: "B" },  // duplicado, fallará
    { _id: 2, nombre: "C" }   // este sí se inserta
  ],
  { ordered: false }
)
🔍 FIND — Leer documentos
mongo shell
// ── find (todos) ──────────────────────────────
db.productos.find()

// ── find con filtro ───────────────────────────
db.productos.find({ categoria: "perifericos" })

// ── findOne ───────────────────────────────────
// Retorna el primer documento que coincida
db.productos.findOne({ nombre: "Mouse Inalámbrico" })

// ── Proyección (2do argumento) ────────────────
// 1 = mostrar | 0 = ocultar
db.productos.find(
  { categoria: "perifericos" },
  { nombre: 1, precio: 1, _id: 0 }
)

// ── Métodos encadenables ──────────────────────
db.productos.find()
  .sort({ precio: -1 })     // -1 desc, 1 asc
  .limit(5)                 // máx 5 resultados
  .skip(10)                 // saltar 10 (paginación)
  .pretty()                  // formato bonito (shell)

// ── Contar ────────────────────────────────────
db.productos.countDocuments({ activo: true })

// Estimación rápida (sin filtro, usa metadatos)
db.productos.estimatedDocumentCount()

// ── Distinct ──────────────────────────────────
// Valores únicos de un campo
db.productos.distinct("categoria")
db.productos.distinct("categoria", { activo: true })
✏️ UPDATE — Actualizar documentos
mongo shell
// ── updateOne ─────────────────────────────────
// Actualiza SOLO el primer documento que coincida
db.productos.updateOne(
  { nombre: "Mouse Inalámbrico" },        // filtro
  { $set: { precio: 40, stock: 180 } }   // cambios
)

// ── updateMany ────────────────────────────────
// Actualiza TODOS los documentos que coincidan
db.productos.updateMany(
  { categoria: "perifericos" },
  { $set: { activo: true } }
)

// ── upsert ────────────────────────────────────
// Si no existe: lo CREA. Si existe: lo ACTUALIZA
db.productos.updateOne(
  { nombre: "Producto Nuevo" },
  { $set: { precio: 99, stock: 10 } },
  { upsert: true }
)

// ── replaceOne ────────────────────────────────
// Reemplaza el documento COMPLETO (excepto _id)
db.productos.replaceOne(
  { nombre: "Cable HDMI" },
  {
    nombre: "Cable HDMI 2.1",
    precio: 25,
    stock: 50,
    categoria: "accesorios"
  }
)

// ── findOneAndUpdate ──────────────────────────
// Actualiza Y retorna el documento
db.productos.findOneAndUpdate(
  { nombre: "SSD 1TB" },
  { $inc: { stock: -1 } },
  {
    returnDocument: "after",   // "before" = doc original
    projection: { nombre: 1, stock: 1 }
  }
)

// ── findOneAndReplace ─────────────────────────
db.productos.findOneAndReplace(
  { nombre: "Hub USB-C" },
  { nombre: "Hub USB-C Pro", precio: 55 },
  { returnDocument: "after" }
)
🗑️ DELETE — Eliminar documentos
mongo shell
// ── deleteOne ─────────────────────────────────
// Elimina el primer documento que coincida
db.productos.deleteOne({ nombre: "Auriculares BT" })

// ── deleteMany ────────────────────────────────
// Elimina TODOS los que coincidan
db.productos.deleteMany({ activo: false })

// Eliminar TODOS los documentos de la colección
db.productos.deleteMany({})

// ── findOneAndDelete ──────────────────────────
// Elimina Y retorna el documento eliminado
db.productos.findOneAndDelete(
  { stock: 0 },
  { projection: { nombre: 1, precio: 1 } }
)

// ── Eliminar la colección completa ────────────
db.productos.drop()

// ── Eliminar múltiples colecciones ───────────
db.clientes.drop()
db.pedidos.drop()
⚠️ Siempre verifica tu filtro con find() antes de ejecutar deleteMany() o updateMany().
02 — Consultas

Filtros y operadores de query

Cómo construir filtros precisos para encontrar exactamente lo que necesitas.

⚖️ Comparación
mongo shell
// Igualdad implícita
db.productos.find({ categoria: "tecnologia" })

// $gt  → mayor que (greater than)
db.productos.find({ precio: { $gt: 100 } })

// $gte → mayor o igual (greater than or equal)
db.productos.find({ precio: { $gte: 100 } })

// $lt  → menor que (less than)
db.productos.find({ precio: { $lt: 50 } })

// $lte → menor o igual
db.productos.find({ precio: { $lte: 50 } })

// $ne  → diferente (not equal)
db.productos.find({ categoria: { $ne: "audio" } })

// Rango combinado: precio entre 50 y 500
db.productos.find({
  precio: { $gte: 50, $lte: 500 }
})

// $in  → está en la lista
db.productos.find({
  categoria: { $in: ["tecnologia", "audio"] }
})

// $nin → NO está en la lista (not in)
db.productos.find({
  categoria: { $nin: ["accesorios", "audio"] }
})
🔗 Lógicos: AND, OR, NOT, NOR
mongo shell
// AND implícito — múltiples condiciones en el mismo objeto
db.productos.find({
  activo: true,
  precio: { $lt: 100 }
})

// $and explícito — útil cuando repites el mismo campo
db.productos.find({
  $and: [
    { activo: true },
    { precio: { $lt: 100 } }
  ]
})

// $or — cumple al menos una condición
db.productos.find({
  $or: [
    { categoria: "audio" },
    { precio: { $gt: 1000 } }
  ]
})

// $nor — NO cumple ninguna de las condiciones
db.productos.find({
  $nor: [
    { activo: false },
    { stock: 0 }
  ]
})

// $not — niega una expresión
db.productos.find({
  precio: { $not: { $gt: 500 } }
})

// Combinación AND + OR
db.productos.find({
  activo: true,
  $or: [
    { categoria: "tecnologia" },
    { rating: { $gte: 4.5 } }
  ]
})
🔎 Existencia, tipo y Regex
mongo shell
// $exists — el campo existe / no existe
db.productos.find({ rating: { $exists: true } })
db.productos.find({ descripcion: { $exists: false } })

// $type — tipo BSON del campo
db.productos.find({ precio: { $type: "double" } })
db.productos.find({ nombre: { $type: "string" } })
db.productos.find({ tags: { $type: "array" } })
// Tipos: string, int, double, bool, date, objectId, array, null, object

// ── REGEX ─────────────────────────────────────
// Forma corta con /patrón/flags
db.productos.find({ nombre: /teclado/i })   // i = case insensitive
db.productos.find({ nombre: /^M/ })          // empieza con M
db.productos.find({ nombre: /HD$/ })          // termina con HD
db.productos.find({ nombre: /\d/ })           // contiene dígito

// Forma con operador $regex
db.productos.find({
  nombre: { $regex: "pro", $options: "i" }
})

// $where — expresión JS (lento, evitar en producción)
db.productos.find({
  $where: "this.precio > this.stock * 5"
})
📊 Proyección y métodos de cursor
mongo shell
// Mostrar solo algunos campos (1 = incluir, 0 = excluir)
db.productos.find(
  {},
  { nombre: 1, precio: 1, _id: 0 }
)

// Excluir campos específicos
db.productos.find(
  {},
  { stock: 0, tags: 0, activo: 0 }
)

// $slice en proyección — primeros/últimos N del array
db.productos.find(
  {},
  { nombre: 1, tags: { $slice: 1 } }     // primer tag
)
db.productos.find(
  {},
  { tags: { $slice: -1 } }               // último tag
)
db.productos.find(
  {},
  { tags: { $slice: [1, 2] } }           // desde índice 1, tomar 2
)

// Métodos del cursor
db.productos.find()
  .sort({ precio: -1, nombre: 1 })         // múltiples campos
  .limit(5)
  .skip(0)
  .projection({ nombre: 1, precio: 1 })

// Paginación: página 3 con 5 elementos por página
db.productos.find()
  .sort({ _id: 1 })
  .skip((3 - 1) * 5)   // (página - 1) * tamaño
  .limit(5)

// forEach — iterar resultados
db.productos.find({ activo: true }).forEach(doc => {
  print(doc.nombre + ": $" + doc.precio)
})
03 — Operadores de Update

Todos los operadores de actualización

Modifica campos sin reescribir el documento completo.

OperadorDescripciónEjemplo de uso
$setAsigna valor (crea el campo si no existe){ $set: { precio: 99 } }
$unsetElimina un campo del documento{ $unset: { descuento: "" } }
$incIncrementa/decrementa un número{ $inc: { stock: -5 } }
$mulMultiplica el valor actual{ $mul: { precio: 1.1 } }
$minActualiza si el nuevo valor es MENOR{ $min: { precio: 50 } }
$maxActualiza si el nuevo valor es MAYOR{ $max: { stock: 200 } }
$renameRenombra un campo{ $rename: { "cat": "categoria" } }
$currentDatePone la fecha/hora actual{ $currentDate: { updatedAt: true } }
$setOnInsertSolo asigna valor si el doc se crea (upsert){ $setOnInsert: { creadoEn: new Date() } }
🔧 Ejemplos de update de campos
mongo shell
// Cambiar precio de un producto
db.productos.updateOne(
  { nombre: "Laptop Pro" },
  { $set: { precio: 1199 } }
)

// Reducir stock al vender (inc con negativo)
db.productos.updateOne(
  { nombre: "SSD 1TB" },
  { $inc: { stock: -5 } }
)

// Varios cambios en una sola operación
db.productos.updateOne(
  { nombre: "Monitor 4K" },
  {
    $set:         { activo: true, categoria: "tecnologia" },
    $inc:         { ventas: 1 },
    $currentDate: { ultimaVenta: true }
  }
)

// Aplicar descuento del 10% a todos los periféricos
db.productos.updateMany(
  { categoria: "perifericos" },
  { $mul: { precio: 0.9 } }
)

// Eliminar un campo
db.productos.updateMany(
  {},
  { $unset: { campo_viejo: "" } }
)

// Renombrar campo en toda la colección
db.productos.updateMany(
  {},
  { $rename: { "cat": "categoria" } }
)

// $min: solo actualiza si el nuevo valor es menor
// (útil para precio mínimo histórico)
db.productos.updateOne(
  { nombre: "SSD 1TB" },
  { $min: { precio_minimo: 80 } }   // solo cambia si 80 < actual
)

// Actualizar campo anidado con dot notation
db.productos.updateOne(
  { nombre: "Laptop Pro" },
  { $set: { "specs.ram": "32GB", "specs.cpu": "i9" } }
)

// $setOnInsert — solo aplica al crear (upsert)
db.productos.updateOne(
  { nombre: "Producto Nuevo" },
  {
    $set:         { precio: 49 },
    $setOnInsert: { creadoEn: new Date(), vistas: 0 }
  },
  { upsert: true }
)
📍 Operador posicional $ en arrays
Modifica un elemento específico del array sin saber su índice.
mongo shell
// $ (posicional) — modifica el PRIMERO que coincide
// El filtro DEBE incluir la condición sobre el array
db.productos.updateOne(
  { tags: "oferta" },
  { $set: { "tags.$": "OFERTA" } }
)

// $[] — modifica TODOS los elementos del array
db.clientes.updateMany(
  {},
  { $set: { "historial.$[].activo": false } }
)

// $[identificador] — filtra qué elementos modificar
db.pedidos.updateMany(
  {},
  { $set: { "items.$[elem].descuento": 0.1 } },
  {
    arrayFilters: [
      { "elem.precio": { $gt: 100 } }   // solo items con precio > 100
    ]
  }
)
04 — Arrays

Operaciones con arrays

MongoDB tiene soporte nativo muy poderoso para arrays dentro de documentos.

🔍 Consultar arrays
mongo shell
// Contiene el elemento "oferta"
db.productos.find({ tags: "oferta" })

// Array EXACTAMENTE igual a este (mismo orden)
db.productos.find({ tags: ["laptop", "oferta"] })

// $all — contiene TODOS estos elementos (cualquier orden)
db.productos.find({
  tags: { $all: ["laptop", "oferta"] }
})

// $size — array de exactamente N elementos
db.productos.find({ tags: { $size: 2 } })

// Acceder por índice (0-based) con dot notation
db.productos.find({ "tags.0": "laptop" })   // primer elemento
db.productos.find({ "tags.1": "oferta" })   // segundo elemento

// $elemMatch — al menos UN elemento cumple TODAS las condiciones
// Útil cuando cada elemento es un objeto
db.pedidos.find({
  items: {
    $elemMatch: {
      precio:   { $gt: 100 },
      cantidad: { $gte: 2 }
    }
  }
})
➕ Modificar arrays
mongo shell
// $push — agrega un elemento al final
db.productos.updateOne(
  { nombre: "Laptop Pro" },
  { $push: { tags: "bestseller" } }
)

// $push con $each — agrega múltiples
db.productos.updateOne(
  { nombre: "Laptop Pro" },
  {
    $push: {
      tags: {
        $each:  ["premium", "nuevo"]
      }
    }
  }
)

// $push + $each + $sort + $slice
// Agrega, ordena y limita el array a los últimos 5
db.productos.updateOne(
  { nombre: "Monitor 4K" },
  {
    $push: {
      historial_precios: {
        $each:  [{ precio: 850, fecha: new Date() }],
        $sort:  { fecha: -1 },
        $slice: 5
      }
    }
  }
)

// $addToSet — push SIN duplicar
db.productos.updateOne(
  { nombre: "Laptop Pro" },
  { $addToSet: { tags: "oferta" } }   // no agrega si ya existe
)

// $addToSet con $each — múltiples, sin duplicar
db.productos.updateOne(
  { nombre: "Laptop Pro" },
  {
    $addToSet: {
      tags: { $each: ["premium", "laptop"] }   // "laptop" ya existe
    }
  }
)

// $pull — elimina elementos que coincidan con la condición
db.productos.updateOne(
  { nombre: "Laptop Pro" },

05 — Aggregation Framework

El Pipeline de Agregación

El corazón de MongoDB. Documentos fluyen de stage en stage, transformándose en cada paso.

ℹ️ ¿Qué es? Un array de etapas. Los documentos entran por el primer stage, se transforman, y el resultado pasa al siguiente. Como una cadena de montaje o una pipe de Linux.
🏭 Flujo visual
Stage 1: $match
Filtra documentos (como WHERE). Solo pasan los que cumplen la condición.
↓
Stage 2: $group
Agrupa y calcula: suma, conteo, promedio, max, min...
↓
Stage 3: $project
Da forma: incluye/excluye campos, crea campos calculados.
↓
Stage 4: $sort
Ordena los resultados.
↓
Stage 5: $limit / $skip
Paginación del resultado final.
🚀 Tu primer pipeline completo
mongo shell
db.productos.aggregate([

  // Stage 1: solo productos activos
  {
    $match: { activo: true }
  },

  // Stage 2: agrupar por categoría y calcular métricas
  {
    $group: {
      _id:            "$categoria",
      totalProductos: { $sum: 1 },
      precioPromedio: { $avg: "$precio" },
      precioMax:      { $max: "$precio" },
      precioMin:      { $min: "$precio" },
      stockTotal:     { $sum: "$stock" },
      ratingPromedio: { $avg: "$rating" }
    }
  },

  // Stage 3: dar formato y renombrar campos
  {
    $project: {
      _id:            0,
      categoria:      "$_id",
      totalProductos: 1,
      stockTotal:     1,
      precioPromedio: { $round: ["$precioPromedio", 2] },
      ratingPromedio: { $round: ["$ratingPromedio", 1] }
    }
  },

  // Stage 4: ordenar por stock total descendente
  {
    $sort: { stockTotal: -1 }
  }

])

Todos los stages importantes

Cada stage tiene su función. Aquí están todos los que debes dominar.

📊 $group — Acumuladores
Agrupa documentos. _id define el criterio. null = agrupa todo junto.
mongo shell
db.productos.aggregate([
  {
    $group: {

      // Criterio de agrupación
      _id: "$categoria",

      // _id: null,                        → todo junto
      // _id: { cat: "$categoria",          → múltiples campos
      //        act: "$activo" },

      // Acumuladores disponibles:
      cantidad:      { $sum: 1 },          // contar docs
      sumaPrecios:   { $sum: "$precio" },   // sumar campo
      promedio:      { $avg: "$precio" },
      maximo:        { $max: "$precio" },
      minimo:        { $min: "$precio" },
      primerNombre:  { $first: "$nombre" },
      ultimoNombre:  { $last: "$nombre" },
      listaNombres:  { $push: "$nombre" },   // array (con repetidos)
      setNombres:    { $addToSet: "$nombre" } // array sin duplicados
    }
  }
])

Reestructura documentos. Incluye, excluye y crea campos calculados.

07 — Joins

$lookup — Unir colecciones

Equivalente al JOIN de SQL. Combina datos de diferentes colecciones en el pipeline.

ℹ️ ¿Cómo funciona? $lookup busca documentos de otra colección que coincidan con un campo, y agrega el resultado como un array en el documento actual. Luego puedes usar $unwind para "aplanarlo".
🔗 $lookup básico
mongo shell
// Traer datos del cliente en cada pedido
db.pedidos.aggregate([

  // Stage 1: hacer el join
  {
    $lookup: {
      from:         "clientes",      // collection a unir
      localField:   "cliente_id",   // campo en pedidos
      foreignField: "email",         // campo en clientes
      as:           "info_cliente"  // nombre del array resultado
    }
  },

  // Stage 2: aplanar el array (1 cliente por pedido)
  {
    $unwind: {
      path:                       "$info_cliente",
      preserveNullAndEmptyArrays: true   // mantener pedidos sin cliente
    }
  },

  // Stage 3: proyectar solo lo necesario
  {
    $project: {
      _id:     0,
      producto: 1,
      total:    1,
      estado:   1,
      cliente:  "$info_cliente.nombre",
      ciudad:   "$info_cliente.ciudad"
    }
  }

])
⚡ $lookup con pipeline (avanzado)
Permite filtrar y transformar la colección secundaria antes de unirla. Mucho más poderoso.
mongo shell
// Para cada cliente, traer sus pedidos activos ordenados por fecha
db.clientes.aggregate([

  {
    $lookup: {
      from: "pedidos",

      // Variables locales disponibles dentro del pipeline
      let: { email_cliente: "$email" },

      pipeline: [

        // Usar $$variable para acceder a la variable local
        {
          $match: {
            $expr: {
              $eq: ["$cliente_id", "$$email_cliente"]
            }
          }
        },

        // Filtrar cancelados
        {
          $match: {
            estado: { $ne: "cancelado" }
          }
        },

        // Ordenar por fecha más reciente
        { $sort: { fecha: -1 } },

        // Solo los campos necesarios
        {
          $project: {
            _id:      0,
            producto: 1,
            total:    1,
            estado:   1
          }
        }

      ],

      as: "pedidos_activos"
    }
  },

  // Solo clientes que tienen al menos un pedido
  {
    $match: {
      "pedidos_activos.0": { $exists: true }
    }
  },

  // Agregar cantidad de pedidos como campo calculado
  {
    $addFields: {
      total_pedidos: { $size: "$pedidos_activos" }
    }
  }

])
🌐 Reporte completo: ventas por ciudad
Pipeline real que combina $lookup + $group + $addFields + $project.
mongo shell
db.pedidos.aggregate([

  // 1. Solo pedidos entregados
  {
    $match: { estado: "entregado" }
  },

  // 2. Unir con clientes para obtener la ciudad
  {
    $lookup: {
      from:         "clientes",
      localField:   "cliente_id",
      foreignField: "email",
      as:           "cliente"
    }
  },
  { $unwind: "$cliente" },

  // 3. Agrupar por ciudad
  {
    $group: {
      _id:         "$cliente.ciudad",
      totalVentas: { $sum: "$total" },
      numPedidos:  { $sum: 1 },
      clientes:    { $addToSet: "$cliente.nombre" }
    }
  },

  // 4. Calcular ticket promedio
  {
    $addFields: {
      ticketPromedio: {
        $divide: ["$totalVentas", "$numPedidos"]
      }
    }
  },

  // 5. Ordenar por total de ventas
  { $sort: { totalVentas: -1 } },

  // 6. Formato final
  {
    $project: {
      _id:            0,
      ciudad:         "$_id",
      totalVentas:    1,
      numPedidos:     1,
      numClientes:    { $size: "$clientes" },
      ticketPromedio: { $round: ["$ticketPromedio", 2] }
    }
  }

])

Índices

Sin índices MongoDB escanea toda la colección. Los índices son la diferencia entre ms y minutos.

⚡ Crear índices
mongo shell
// ── Índice simple ─────────────────────────────
// 1 = ascendente | -1 = descendente
db.productos.createIndex({ precio: 1 })
db.productos.createIndex({ nombre: -1 })

db.productos.createIndex({ categoria: 1, precio: -1 })

db.clientes.createIndex(

db.productos.createIndex(

db.sesiones.createIndex(

📖 Tipos de índice — Explicación completa
MongoDB tiene 5 tipos principales de índice. Cada uno sirve para un caso de uso diferente.
1. ÍNDICE DE TEXTO
Permite hacer búsquedas de palabras dentro de campos string. Ignora mayúsculas/minúsculas y palabras comunes ("el", "la", "de"). Ideal para buscadores. Solo puede haber uno por colección, pero puede cubrir varios campos.
2. ÍNDICE DE ARREGLOS (Multikey)
Se crea automáticamente cuando indexas un campo que contiene un array. MongoDB indexa cada elemento del array por separado, así puedes buscar por cualquier elemento eficientemente.
3. UNIQUE INDEX
Garantiza que ningún documento repita el mismo valor en ese campo. Si intentas insertar un duplicado, MongoDB lanza un error. Útil para emails, cédulas, ISBN, usernames.
4. ÍNDICE DE RANGO
Es el índice estándar (el que más usas). Ordena los valores en un B-Tree, lo que hace muy eficientes las consultas con $gt, $lt, $between, rangos de fechas, etc.
5. ÍNDICE GEOESPACIAL
Para datos de ubicación (latitud/longitud). 2dsphere funciona con la curvatura de la Tierra (para coordenadas reales). 2d para planos cartesianos (mapas planos, videojuegos). Permite consultas como "restaurantes a menos de 2km de aquí".
mongo shell — los 5 tipos
// ─── 1. ÍNDICE DE TEXTO ───────────────────────────────────
// Busca palabras dentro de strings. Solo 1 por colección.
db.libros.createIndex(
  { titulo: "text", descripcion: "text" }
)
// Uso: buscar la palabra "soledad" en título o descripción
db.libros.find({ $text: { $search: "soledad" } })
// Excluir palabra:
db.libros.find({ $text: { $search: "novela -romance" } })
// Con relevancia (score):
db.libros.find(
  { $text: { $search: "cien años" } },
  { score: { $meta: "textScore" } }
).sort({ score: { $meta: "textScore" } })


// ─── 2. ÍNDICE DE ARREGLOS (Multikey) ────────────────────
// Se crea igual que un índice normal sobre un campo array.
// MongoDB automáticamente indexa CADA elemento del array.
db.libros.createIndex({ autores: 1 })
// Uso: buscar libros de un autor específico dentro del array
db.libros.find({ autores: ObjectId("698cc2010d7212fce1af8a03") })
// También funciona con $in sobre arrays:
db.libros.find({ tags: { $in: ["ciencia", "ficcion"] } })
// NOTA: no se pueden combinar dos campos multikey en un índice compuesto


// ─── 3. UNIQUE INDEX ──────────────────────────────────────
// No permite valores repetidos en ese campo.
db.clientes.createIndex(
  { email: 1 },
  { unique: true }
)
// Insertar email duplicado → lanza error E11000 duplicate key
// Índice único compuesto (la COMBINACIÓN debe ser única):
db.productos.createIndex(
  { nombre: 1, categoria: 1 },
  { unique: true }
)
// Unique + sparse: permite múltiples null (no los considera duplicados)
db.clientes.createIndex(
  { telefono: 1 },
  { unique: true, sparse: true }
)


// ─── 4. ÍNDICE DE RANGO ───────────────────────────────────
// El índice estándar. Perfecto para $gt, $lt, $gte, $lte, rangos.
db.productos.createIndex({ precio: 1 })
// Uso con rango de precios:
db.productos.find({ precio: { $gte: 100, $lte: 500 } })
// Índice de rango en fechas (muy común):

    $gte: ISODate("2024-01-01"),

  }
})


// ─── 5. ÍNDICE GEOESPACIAL ────────────────────────────────
// Para ubicaciones. Los documentos deben tener formato GeoJSON.
// Insertar un lugar con coordenadas:
db.lugares.insertOne({
  nombre: "Biblioteca Central",
  ubicacion: {
    type: "Point",

  ubicacion: {
    $near: {
      $geometry: {
        type: "Point",
        coordinates: [-75.5636, 6.2518]

    }

})

db.lugares.find({
  ubicacion: {

{
  $setWindowFields: {
    partitionBy: "$categoria",  // como GROUP BY (opcional)
    sortBy:      { precio: -1 }, // como ORDER BY
    output: {
      // aquí defines los campos calculados
    }
  }
}

db.productos.aggregate([
  {
    $setWindowFields: {
      partitionBy: "$categoria",
      sortBy:      { precio: -1 },
      output: {
        ranking: {
          $rank: {}   // posición dentro de su categoría
        }
      }
    }
  }
])
// Resultado: cada producto tiene su ranking dentro de su categoría

// ─── Ejemplo 2: Total acumulado de ventas por fecha ───
db.pedidos.aggregate([
  {
    $setWindowFields: {
      partitionBy: null,
      sortBy: { fecha: 1 },
      output: {
        ventasAcumuladas: {
          $sum: "$total",
          window: {
            documents: ["unbounded", "current"]
            // desde el inicio hasta el documento actual
          }
        }
      }
    }
  }
])

// ─── Ejemplo 3: Diferencia con el documento anterior ──
db.pedidos.aggregate([
  { $sort: { fecha: 1 } },
  {
    $setWindowFields: {
      sortBy: { fecha: 1 },
      output: {
        totalAnterior: {
          $shift: {
            output: "$total",
            by:     -1,       // -1 = documento anterior
            default: 0
          }
        }
      }
    }
  },
  {
    $addFields: {
      diferencia: { $subtract: ["$total", "$totalAnterior"] }
    }
  }
])

// ─── Operadores de window disponibles ─────────────────
// $sum, $avg, $min, $max  → igual que en $group
// $count                  → cuenta en la ventana
// $rank                   → posición (con saltos en empates)
// $denseRank              → posición (sin saltos en empates)
// $documentNumber         → número de fila secuencial
// $shift                  → valor del doc anterior/siguiente
// $first / $last          → primer/último en la ventana
